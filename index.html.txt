        <div class="btn" id="btn-d">D</div>
    </div>

    <div id="action-pads">
        <div class="action-btn" id="btn-build">BAUEN</div>
        <div class="action-btn" id="btn-destroy">ABBAUEN</div>

    <script>
        // 1. SCHAUPLATZ & KAMERA EINRICHTEN
        const scene = new THREE.Scene();
        scene.background = new THREE.Color(0x6eb7ff); // Schönerer Himmel
        scene.fog = new THREE.FogExp2(0x6eb7ff, 0.015); // Realistischer Nebel in der Ferne

        const camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);
        const renderer = new THREE.WebGLRenderer({ antialias: true }); // Antialiasing für glatte Kanten
        renderer.setSize(window.innerWidth, window.innerHeight);
        renderer.shadowMap.enabled = true; // SCHATTEN AKTIVIEREN!
        renderer.shadowMap.type = THREE.PCFSoftShadowMap;
        document.body.appendChild(renderer.domElement);

        // 2. REALISTISCHES LICHT (SONNE)
        const ambientLight = new THREE.AmbientLight(0xffffff, 0.4); // Grundhelligkeit
        scene.add(ambientLight);

        const sunLight = new THREE.DirectionalLight(0xffffff, 0.8); // Die Sonne
        sunLight.position.set(40, 60, 20);
        sunLight.castShadow = true; // Sonne wirft Schatten
        sunLight.shadow.mapSize.width = 2048; // Hochauflösende Schatten
        sunLight.shadow.mapSize.height = 2048;
        sunLight.shadow.camera.near = 0.5;
        sunLight.shadow.camera.far = 150;
        const d = 40;
        sunLight.shadow.camera.left = -d;
        sunLight.shadow.camera.right = d;
        sunLight.shadow.camera.top = d;
        sunLight.shadow.camera.bottom = -d;
        scene.add(sunLight);

        // 3. TEXTUREN GENERIEREN (Falls keine Bilder da sind, erzeugen wir realistische Shader-Materialien)
        const createBlockMaterial = (colorHex) => {
            return new THREE.MeshStandardMaterial({ 
                color: colorHex, 
                roughness: 0.8, // Nicht spiegelnd wie Plastik
                metalness: 0.1 
            });
        };

        const grassMaterial = createBlockMaterial(0x557a2b);
        const dirtMaterial = createBlockMaterial(0x866043);
        const stoneMaterial = createBlockMaterial(0x737373);

        // 4. EINE WELT GENERIEREN (Hügelig statt flach!)
        const blocks = [];
        const worldSize = 20;

        for (let x = -worldSize/2; x < worldSize/2; x++) {
            for (let z = -worldSize/2; z < worldSize/2; z++) {
                // Mathematische Formel für hügeliges Gelände (Simplex-Ersatz)
                const height = Math.floor((Math.sin(x * 0.2) + Math.cos(z * 0.2)) * 2) + 3;
                
                for (let y = 0; y <= height; y++) {
                    let material = dirtMaterial;
                    if (y === height) material = grassMaterial; // Oben drauf Gras
                    if (y < height - 2) material = stoneMaterial; // Tief unten Stein

                    const geometry = new THREE.BoxGeometry(1, 1, 1);
                    const block = new THREE.Mesh(geometry, material);
                    block.position.set(x, y, z);
                    block.castShadow = true; // Block wirft Schatten
                    block.receiveShadow = true; // Block empfängt Schatten
                    scene.add(block);
                    blocks.push(block);
                }
            }
        }

        // Spieler-Startposition
        camera.position.set(0, 8, 12);
        camera.lookAt(0, 4, 0);

        // 5. STEUERUNG & INTERAKTION
        let move = { w: false, a: false, s: false, d: false };
        const speed = 0.15;

        // Touch-Events verknüpfen
        const bindTouch = (id, direction) => {
            const btn = document.getElementById(id);
            btn.addEventListener('touchstart', (e) => { e.preventDefault(); move[direction] = true; });
            btn.addEventListener('touchend', () => move[direction] = false);
        };
        bindTouch('btn-w', 'w'); bindTouch('btn-a', 'a'); bindTouch('btn-s', 's'); bindTouch('btn-d', 'd');

        // BAUEN UND ABBAUEN
        let currentMode = 'build';
        document.getElementById('btn-build').addEventListener('touchstart', () => currentMode = 'build');
        document.getElementById('btn-destroy').addEventListener('touchstart', () => currentMode = 'destroy');

        // Klicken/Tippen in der Welt
        window.addEventListener('pointerdown', (event) => {
            // Verhindern, dass Klicks auf UI-Buttons Blöcke setzen
            if(event.clientY > window.innerHeight - 120 && (event.clientX < 220 || event.clientX > window.innerWidth - 180)) return;

            const mouse = new THREE.Vector2(
                (event.clientX / window.innerWidth) * 2 - 1,
                -(event.clientY / window.innerHeight) * 2 + 1
            );

            const raycaster = new THREE.Raycaster();
            raycaster.setFromCamera(mouse, camera);
            const intersects = raycaster.intersectObjects(blocks);

            if (intersects.length > 0) {
                const hit = intersects[0];
                
                if (currentMode === 'destroy') {
                    // BLOCK LÖSCHEN
                    scene.remove(hit.object);
                    const index = blocks.indexOf(hit.object);
                    if (index > -1) blocks.splice(index, 1);
                } else {
                    // BLOCK BAUEN (An der exakten Trefferseite)
                    const position = hit.object.position.clone().add(hit.face.normal);
                    const geometry = new THREE.BoxGeometry(1, 1, 1);
                    const newBlock = new THREE.Mesh(geometry, stoneMaterial); // Neue Blöcke sind aus Stein
                    newBlock.position.copy(position);
                    newBlock.castShadow = true;
                    newBlock.receiveShadow = true;
                    scene.add(newBlock);
                    blocks.push(newBlock);
                }
            }
        });

        // 6. GAME LOOP (ANMATION)
        function animate() {
            requestAnimationFrame(animate);

            // Bewegung umsetzen
            if (move.w) camera.position.z -= speed;
            if (move.s) camera.position.z += speed;
            if (move.a) camera.position.x -= speed;
            if (move.d) camera.position.x += speed;

            renderer.render(scene, camera);
        }
        animate();

        // Screen-Resize absichern
        window.addEventListener('resize', () => {
            camera.aspect = window.innerWidth / window.innerHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth, window.innerHeight);
        });
    </script>
