    <style>
        body { margin: 0; overflow: hidden; font-family: sans-serif; user-select: none; }
        
        /* Touch-Steuerung für das Handy */
        #controls {
            position: absolute; bottom: 20px; left: 20px;
            display: grid; grid-template-columns: repeat(3, 50px); grid-template-rows: repeat(3, 50px);
            gap: 5px; z-index: 10;
        }
        .btn {
            background: rgba(255,255,255,0.5); border: 2px solid #fff; border-radius: 10px;
            display: flex; align-items: center; justify-content: center; font-weight: bold; font-size: 18px;
        }
        .btn:active { background: rgba(255,255,255,0.8); }
        #btn-w { grid-column: 2; grid-row: 1; }
        #btn-a { grid-column: 1; grid-row: 2; }
        #btn-s { grid-column: 2; grid-row: 2; }
        #btn-d { grid-column: 3; grid-row: 2; }

        #btn-action {
            position: absolute; bottom: 20px; right: 20px;
            width: 70px; height: 70px; background: rgba(255, 0, 0, 0.6);
            color: white; border-radius: 50%; display: flex; align-items: center; justify-content: center;
            font-weight: bold; z-index: 10; border: 2px solid white;
        }
    </style>
    <script src="https://cdnjs.cloudflare.com/ajax/libs/three.js/r128/three.min.js"></script>

<body>

    <div id="controls">
        <div class="btn" id="btn-w">W</div>
        <div class="btn" id="btn-a">A</div>
        <div class="btn" id="btn-s">S</div>
        <div class="btn" id="btn-d">D</div>
    </div>
    <div id="btn-action">BAUEN</div>

    <script>
        const scene = new THREE.Scene();
        scene.background = new THREE.Color(0x87CEEB);
        
        const camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);
        const renderer = new THREE.WebGLRenderer({ antialias: true });
        renderer.setSize(window.innerWidth, window.innerHeight);
        document.body.appendChild(renderer.domElement);

        const light = new THREE.AmbientLight(0x404040, 2);
        scene.add(light);
        const dirLight = new THREE.DirectionalLight(0xffffff, 0.8);
        dirLight.position.set(10, 20, 15);
        scene.add(dirLight);

        // Bodenplatten
        const geometry = new THREE.BoxGeometry(1, 1, 1);
        const material = new THREE.MeshLambertMaterial({ color: 0x55aa44 });
        for(let x = -5; x < 5; x++) {
            for(let z = -5; z < 5; z++) {
                const mesh = new THREE.Mesh(geometry, material);
                mesh.position.set(x, 0, z);
                scene.add(mesh);
            }
        }

        camera.position.set(0, 2, 5);

        // Bewegungsvariablen
        let move = { forward: false, backward: false, left: false, right: false };
        let rotY = 0;

        // Touch-Events für Bewegungknöpfe
        const setupBtn = (id, direction) => {
            const el = document.getElementById(id);
            el.addEventListener('touchstart', (e) => { e.preventDefault(); move[direction] = true; });
            el.addEventListener('touchend', () => move[direction] = false);
        };
        setupBtn('btn-w', 'forward');
        setupBtn('btn-s', 'backward');
        setupBtn('btn-a', 'left');
        setupBtn('btn-d', 'right');

        // Umschauen durch Wischen auf dem Bildschirm
        let touchStart = { x: 0, y: 0 };
        document.addEventListener('touchstart', (e) => {
            if(e.target.className !== 'btn' && e.target.id !== 'btn-action') {
                touchStart.x = e.touches[0].clientX;
            }
        }, {passive: true});

        document.addEventListener('touchmove', (e) => {
            if(e.target.className !== 'btn' && e.target.id !== 'btn-action') {
                let deltaX = e.touches[0].clientX - touchStart.x;
                rotY -= deltaX * 0.005;
                touchStart.x = e.touches[0].clientX;
            }
        }, {passive: true});

        // Block bauen Button
        document.getElementById('btn-action').addEventListener('touchstart', (e) => {
            e.preventDefault();
            const newBlock = new THREE.Mesh(geometry, new THREE.MeshLambertMaterial({ color: 0x888888 }));
            newBlock.position.set(
                Math.round(camera.position.x - Math.sin(rotY) * 2),
                Math.round(camera.position.y - 0.5),
                Math.round(camera.position.z - Math.cos(rotY) * 2)
            );
            scene.add(newBlock);
        });

        // Game Loop
        const speed = 0.06;
        function animate() {
            requestAnimationFrame(animate);

            camera.rotation.set(0, rotY, 0);

            if(move.forward) { camera.position.x -= Math.sin(rotY) * speed; camera.position.z -= Math.cos(rotY) * speed; }
            if(move.backward) { camera.position.x += Math.sin(rotY) * speed; camera.position.z += Math.cos(rotY) * speed; }
            if(move.left) { camera.position.x -= Math.cos(rotY) * speed; camera.position.z += Math.sin(rotY) * speed; }
            if(move.right) { camera.position.x += Math.cos(rotY) * speed; camera.position.z -= Math.sin(rotY) * speed; }

            renderer.render(scene, camera);
        }
        animate();

        window.addEventListener('resize', () => {
            camera.aspect = window.innerWidth / window.innerHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth, window.innerHeight);
        });
    </script>
